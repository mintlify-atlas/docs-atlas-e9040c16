  <div className="mt-16 mb-16 max-w-5xl mx-auto px-6">
    <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick start</h2>
    <p className="text-base text-gray-600 dark:text-gray-400 mb-8">Get up and running with the Linkvite Go SDK in minutes</p>
    
    <Steps>
      <Step title="Install the SDK">
        Install the Linkvite Go SDK using `go get`:
        
        ```bash
        go get github.com/linkvite/go
        ```
      </Step>
      
      <Step title="Get your API key">
        Sign up for a Linkvite account and generate an API key from your dashboard at [linkvite.io](https://linkvite.io).
        
        <Note>
          API keys start with `link_` and provide secure access to your Linkvite account.
        </Note>
      </Step>
      
      <Step title="Create your first client">
        Initialize the client with your API key:
        
        ```go
        package main

        import (
            "context"
            "fmt"
            "log"

            linkvite "github.com/linkvite/go"
        )

        func main() {
            client, err := linkvite.NewClient("link_your-api-key")
            if err != nil {
                log.Fatal(err)
            }
            
            ctx := context.Background()
            
            // Get current user
            user, err := client.User.Get(ctx)
            if err != nil {
                log.Fatal(err)
            }
            fmt.Printf("Hello, %s!\n", user.Name)
        }
        ```
      </Step>
      
      <Step title="Create your first bookmark">
        Save a bookmark to your Linkvite account:
        
        ```go
        bookmark, err := client.Bookmarks.Create(ctx, &linkvite.CreateBookmarkInput{
            URL:         "https://example.com",
            Title:       "Example Site",
            Description: "An example website",
        })
        if err != nil {
            log.Fatal(err)
        }
        fmt.Printf("Created bookmark: %s\n", bookmark.Title)
        ```
        
        <Accordion title="Example response">
          The API returns a complete `Bookmark` object with metadata:
          
          ```json
          {
            "id": "bmk_abc123",
            "title": "Example Site",
            "description": "An example website",
            "url": "https://example.com",
            "created_at": "2024-01-15T10:30:00Z"
          }
          ```
        </Accordion>
      </Step>
    </Steps>
